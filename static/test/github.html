TEST_CASE("json ") {
    struct A {
        bool a;
        std::vector<std::vector<std::string>> b;
        std::unordered_map<std::string, std::string> c;
        std::optional<std::string> d;
        std::shared_ptr<std::string> e;
        double f;
    } t{};

    reflection::fromJson(t, R"({
        "f": -3.1415926E+10,
        "d": null,
        "a": 0,
        "c": {
            "": "",
            "Key": "Value",
            "!@#$%^&*()_+": "Value",
            "": "\n",
            "": "\tTabTest"
        },
        "b": [
            [],
            [""],
            ["\n", "\"\\\'", "", "\u4E2D\u6587"]
        ],
        "e": "HelloPtr"
    })");

    // 
    REQUIRE(t.a == false);
    REQUIRE(t.f == -3.1415926E+10);
    REQUIRE(!t.d.has_value());
    REQUIRE(t.e != nullptr);

    REQUIRE(t.c.at("") == "");
    REQUIRE(t.c.at("Key") == "Value");
    REQUIRE(t.c.at("!@#$%^&*()_+") == "Value");
